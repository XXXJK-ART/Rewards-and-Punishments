rp_add_merit_effect = {
	save_scope_as = rp_merit_recipient

	if = {
		limit = { exists = $LIEGE$ }
		set_variable = {
			name = rp_merit_owner
			value = $LIEGE$
		}
		$LIEGE$ = {
			if = {
				limit = {
					NOT = {
						is_target_in_variable_list = {
							name = rp_merit_roster
							target = scope:rp_merit_recipient
						}
					}
				}
				add_to_variable_list = {
					name = rp_merit_roster
					target = scope:rp_merit_recipient
				}
			}
		}
	}

	if = {
		limit = { NOT = { has_variable = rp_merit_score } }
		set_variable = {
			name = rp_merit_score
			value = 0
		}
	}
	change_variable = {
		name = rp_merit_score
		add = $VALUE$
	}
	if = {
		limit = { var:rp_merit_score > 200 }
		set_variable = {
			name = rp_merit_score
			value = 200
		}
	}
	if = {
		limit = { var:rp_merit_score < -100 }
		set_variable = {
			name = rp_merit_score
			value = -100
		}
	}
	rp_update_great_house_status_effect = yes
}

rp_consume_merit_effect = {
	if = {
		limit = { NOT = { has_variable = rp_merit_score } }
		set_variable = {
			name = rp_merit_score
			value = 0
		}
	}
	change_variable = {
		name = rp_merit_score
		add = -$VALUE$
	}
	if = {
		limit = { var:rp_merit_score < 0 }
		set_variable = {
			name = rp_merit_score
			value = 0
		}
	}
}

		rp_add_merit_effect = { LIEGE = root VALUE = $VALUE$ }
		rp_add_founding_seniority_effect = { VALUE = 2 }
		add_opinion = {
			target = root
			modifier = rp_campaign_incident_grateful_opinion
		}
	}
	add_prestige = $PRESTIGE$
	rp_mark_campaign_actor_used_effect = yes
}

rp_apply_campaign_failure_effect = {
	scope:rp_campaign_actor = {
		rp_add_merit_effect = { LIEGE = root VALUE = $VALUE$ }
		rp_add_founding_seniority_effect = { VALUE = 1 }
		add_opinion = {
			target = root
			modifier = rp_campaign_incident_rebuked_opinion
		}
		add_character_modifier = {
			modifier = rp_merit_resentful_modifier
			years = 3
		}
		add_character_flag = rp_resentful_mark
	}
	add_prestige = $PRESTIGE$
	rp_mark_campaign_actor_used_effect = yes
}

rp_apply_campaign_risk_effect = {
	scope:rp_campaign_actor = {
		add_opinion = {
			target = root
			modifier = rp_campaign_incident_risk_opinion
		}
		add_character_flag = {
			flag = rp_campaign_incident_recent_actor
			days = 180
		}
	}
}

rp_apply_defection_effect = {
	scope:rp_campaign_actor = {
		add_character_modifier = {
			modifier = rp_traitor_modifier
			years = 10
		}
		add_opinion = {
			target = root
			modifier = rp_merit_refused_harsh_opinion
		}
		rp_clear_merit_effect = yes
	}
	add_prestige = -150
	add_character_modifier = {
		modifier = rp_campaign_disorder_modifier
		years = 3
	}
	if = {
		limit = {
			exists = $WAR$
			scope:rp_campaign_actor = { is_landed = yes }
			$WAR$ = { primary_attacker = { this = root } }
		}
		$WAR$ = {
			remove_participant = scope:rp_campaign_actor
			add_defender = scope:rp_campaign_actor
		}
	}
	else_if = {
		limit = {
			exists = $WAR$
			scope:rp_campaign_actor = { is_landed = yes }
			$WAR$ = { primary_defender = { this = root } }
		}
		$WAR$ = {
			remove_participant = scope:rp_campaign_actor
			add_attacker = scope:rp_campaign_actor
		}
	}
	rp_mark_campaign_actor_used_effect = yes
}
